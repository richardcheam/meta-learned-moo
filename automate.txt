#!/bin/bash

DATASETS=("adult" "compas" "credit" "mnist" "fashion" "electricity")
OPTIMS=("adam" "velo")
SEEDS=(0 1 2)
EPOCHS=100
SOLVER="epo"

for DATASET in "${DATASETS[@]}"; do
  for OPTIM in "${OPTIMS[@]}"; do
    for SEED in "${SEEDS[@]}"; do
      echo "Running $DATASET $OPTIM seed=$SEED"
      ./run-phn-velo.sh "$DATASET" "$OPTIM" "$SOLVER" "$SEED" "$EPOCHS"
    done
  done
done
